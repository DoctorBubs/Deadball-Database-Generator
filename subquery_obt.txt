SELECT players.player_name,players.pos,players.hand, players.obt,teams.team_name
FROM players
INNER JOIN teams ON teams.id = players.team_id
WHERE players.obt > (
    SELECT AVG(obt) FROM players WHERE players.PD IS NULL )
AND players.PD IS NULL
ORDER BY players.obt DESC;
